Practical 4: String Concepts

Task / Question
4.To study string concepts, escape characters, string formatting, raw strings, Unicode strings, and built-in string methods.
Python 

# String Concepts & Escape Characters
str1 = "Hello\nWorld"  # \n for newline
str2 = "Python\tProgramming"  # \t for tab
str3 = "He said \"Hi!\""  # \" for quotes

print("Escape Characters:")
print(str1)
print(str2)
print(str3)

# String Formatting
name = "DARPAN"
age = 20
print("\nString Formatting:")
print("My name is {} and I am {} years old.".format(name, age))
print(f"My name is {name} and I am {age} years old.")  # f-string

# Raw Strings
raw_str = r"C:\Users\DARPAN\Documents"
print("\nRaw String:", raw_str)

# Unicode Strings
unicode_str = "Unicode example: \u2764 \u263A"
print("\nUnicode String:", unicode_str)

# Built-in String Methods
sample = "python programming"
print("\nBuilt-in String Methods:")
print("Uppercase:", sample.upper())
print("Lowercase:", sample.lower())
print("Title Case:", sample.title())
print("Count 'p':", sample.count('p'))
print("Starts with 'py':", sample.startswith("py"))
print("Ends with 'ing':", sample.endswith("ing"))
print("Find 'gram':", sample.find("gram"))
print("Replace 'python' with 'Java':", sample.replace("python", "Java"))
