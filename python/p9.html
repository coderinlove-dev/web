Practical 9: Exception Handling

Task / Question
9.To study Python exception handling including assert statement, multiple except clauses, try-finally, raising exceptions, and user-defined exceptions.
Python 

# Assert Statement
num = 10
assert num > 0, "Number must be positive"

# Multiple Except Clauses
try:
    a = int(input("Enter first number: "))
    b = int(input("Enter second number: "))
    result = a / b
    print("Division Result:", result)
except ZeroDivisionError:
    print("Error: Division by zero")
except ValueError:
    print("Error: Invalid input")
except Exception as e:
    print("Error:", e)

# Try-Finally
try:
    f = open("testfile.txt", "w")
    f.write("Python Exception Handling")
finally:
    f.close()
    print("File closed successfully")

# Raising Exception
def check_age(age):
    if age < 18:
        raise ValueError("Age must be 18 or above")
    else:
        print("Age is valid")

try:
    check_age(16)
except ValueError as ve:
    print("Raised Exception:", ve)

# User-defined Exception
class NegativeNumberError(Exception):
    pass

def square_root(n):
    if n < 0:
        raise NegativeNumberError("Cannot compute square root of negative number")
    else:
        print("Square root:", n ** 0.5)

try:
    square_root(-25)
except NegativeNumberError as ne:
    print("User-defined Exception:", ne)
