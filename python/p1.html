Task / Question
1.To understand Python identifiers, input/output, data types, type conversion, operators, and mathematical functions.
Python 

import math

# Identifiers and Variables
num1 = int(input("Enter first number: "))
num2 = float(input("Enter second number: "))

# Display Data Types
print("Data type of num1:", type(num1))
print("Data type of num2:", type(num2))

# Type Conversion
sum_value = num1 + int(num2)
print("After type conversion (int):", sum_value)

# Operators
print("Addition:", num1 + num2)
print("Subtraction:", num1 - num2)
print("Multiplication:", num1 * num2)
print("Division:", num1 / num2)
print("Modulus:", num1 % int(num2))
print("Exponent:", num1 ** 2)

# Mathematical Functions
print("Square root of num1:", math.sqrt(num1))
print("Power (num1^3):", math.pow(num1, 3))
print("Absolute value:", abs(-num1))
