Practical 6: Functional Programming Tools

Task / Question
6.To implement functional programming tools (filter, map, reduce), lists as stacks and queues, and list comprehensions
Python 

from functools import reduce

# Functional Programming Tools
numbers = [1, 2, 3, 4, 5, 6]

# Filter: even numbers
even_numbers = list(filter(lambda x: x % 2 == 0, numbers))
print("Even Numbers (filter):", even_numbers)

# Map: square of numbers
squared_numbers = list(map(lambda x: x ** 2, numbers))
print("Squared Numbers (map):", squared_numbers)

# Reduce: product of numbers
product = reduce(lambda x, y: x * y, numbers)
print("Product of Numbers (reduce):", product)

# Lists as Stack (LIFO)
stack = []
stack.append(10)
stack.append(20)
stack.append(30)
print("\nStack after pushes:", stack)
stack.pop()
print("Stack after pop:", stack)

# Lists as Queue (FIFO)
queue = []
queue.append(1)
queue.append(2)
queue.append(3)
print("\nQueue after enqueues:", queue)
queue.pop(0)
print("Queue after dequeue:", queue)

# List Comprehensions
squares = [x ** 2 for x in range(1, 6)]
print("\nSquares using list comprehension:", squares)

even_sq = [x ** 2 for x in range(1, 11) if x % 2 == 0]
print("Squares of even numbers using list comprehension:", even_sq)
