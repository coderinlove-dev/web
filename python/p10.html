Practical 10: Class And Subclass

Task / Question
10.To create a class and subclass, demonstrate data hiding, class variables, method overriding, and basic exception handling.
Python 

# Base Class
class Employee:
    company = "TechCorp"  # Class Variable
    
    def __init__(self, name, salary):
        self.name = name       # Public
        self.__salary = salary # Private (data hiding)
    
    def display(self):
        print(f"Employee Name: {self.name}, Salary: {self.__salary}, Company: {Employee.company}")
    
    # Method to update salary
    def update_salary(self, amount):
        if amount < 0:
            raise ValueError("Salary increment cannot be negative")
        self.__salary += amount

# Subclass
class Manager(Employee):
    def __init__(self, name, salary, department):
        super().__init__(name, salary)
        self.department = department
    
    # Method Overriding
    def display(self):
        print(f"Manager Name: {self.name}, Department: {self.department}, Company: {Employee.company}")

# Creating Objects
e1 = Employee("Anup", 50000)
m1 = Manager("Soham", 80000, "IT")

e1.display()
m1.display()

# Exception Handling while updating salary
try:
    e1.update_salary(-5000)
except ValueError as ve:
    print("Exception:", ve)

# Valid salary update
e1.update_salary(10000)
e1.display()
