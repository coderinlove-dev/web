Practical 8: Object Oriented Programming

Task / Question
8.To study object-oriented programming concepts in Python including classes, objects, data hiding, constructors, class variables, method overloading, operator overloading, inheritance, and method overriding.
Python 

# Class and Constructor
class Person:
    # Class Variable
    species = "Janavar "
    
    def __init__(self, name, age):
        self.name = name  # Public
        self.__age = age  # Private (data hiding)
    
    # Method
    def display(self):
        print(f"Name: {self.name}, Age: {self.__age}, Species: {Person.species}")
    
    # Method Overloading (simulated using default args)
    def greet(self, msg="Hello"):
        print(f"{msg}, {self.name}!")

    # Operator Overloading
    def __add__(self, other):
        return self.__age + other.__age

# Creating Objects
p1 = Person("Omkar", 22)
p2 = Person("Rahul", 21)

p1.display()
p2.display()

p1.greet()
p2.greet("Good Morning")

# Accessing Private Variable (will cause error if uncommented)
# print(p1.__age)

# Operator Overloading
print("Sum of ages (operator +):", p1 + p2)

# Inheritance
class Student(Person):
    def __init__(self, name, age, roll_no):
        super().__init__(name, age)
        self.roll_no = roll_no
    
    # Method Overriding
