Practical 3: String Operations & Random Functions

Task / Question
3.To implement string operations, random functions, command line arguments, and operator precedence.
Python 

import random
import sys

# String Operations
str1 = input("Enter first string: ")
str2 = input("Enter second string: ")

print("Concatenation:", str1 + str2)
print("Repetition:", str1 * 2)
print("Length of str1:", len(str1))
print("Uppercase str1:", str1.upper())
print("Lowercase str2:", str2.lower())
print("Is str1 alphabetic?", str1.isalpha())

# Random Functions
print("\nRandom Numbers:")
print("Random float [0,1):", random.random())
print("Random integer [1,10]:", random.randint(1, 10))
print("Random choice from list:", random.choice([1, 2, 3, 4, 5]))

# Command Line Arguments
# Example: python program.py arg1 arg2
print("\nCommand Line Arguments:")
print("Total arguments:", len(sys.argv))
print("Arguments List:", sys.argv)

# Operator Precedence
a = 10
b = 5
c = 2
result = a + b * c / (b - c)
print("\nOperator Precedence Result (a + b * c / (b - c)):", result)
