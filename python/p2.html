Practical 2: Control Statements & Loops

Task / Question
2.To implement control statements, loops, membership operators, identity operators, and loop control statements.
Python 

# Control Statement (if-else)
num = int(input("Enter a number: "))

if num > 0:
    print("Number is Positive")
elif num == 0:
    print("Number is Zero")
else:
    print("Number is Negative")

# Loop (for loop)
print("\nFor Loop:")
for i in range(1, 6):
    print(i, end=" ")

# Loop (while loop)
print("\n\nWhile Loop:")
count = 1
while count <= 5:
    print(count, end=" ")
    count += 1

# Membership Operators
list1 = [10, 20, 30, 40]
print("\n\nMembership Operators:")
print("20 in list1:", 20 in list1)
print("50 not in list1:", 50 not in list1)

# Identity Operators
a = [1, 2, 3]
b = a
c = [1, 2, 3]

print("\nIdentity Operators:")
print("a is b:", a is b)
print("a is c:", a is c)
print("a is not c:", a is not c)

# Loop Control Statements
print("\nLoop Control Statements (break & continue):")
for i in range(1, 6):
    if i == 3:
        continue
    if i == 5:
        break
    print(i, end=" ")
