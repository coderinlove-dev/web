Practical 5: Creating , Accesing and Deleting

Task / Question
5. To study list creation, accessing, updating, deleting elements, slicing, reverse indexing, matrices, and built-in list functions.
Python 

# List Creation
list1 = [10, 20, 30, 40, 50]
matrix = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]

print("Original List:", list1)
print("Matrix:")
for row in matrix:
    print(row)

# Accessing Elements
print("\nAccessing Elements:")
print("First element:", list1[0])
print("Last element (reverse indexing):", list1[-1])
print("Matrix element [1][2]:", matrix[1][2])

# Updating Elements
list1[2] = 35
print("\nAfter Updating 3rd element:", list1)

# Deleting Elements
del list1[1]  # Delete 2nd element
print("After Deleting 2nd element:", list1)

# Slicing
print("\nSlicing:")
print("First 3 elements:", list1[:3])
print("Last 2 elements:", list1[-2:])

# Reverse Indexing
print("\nReverse Indexing:")
print("Last element:", list1[-1])
print("Second last element:", list1[-2])

# Built-in List Functions
print("\nBuilt-in List Functions:")
list1.append(60)
print("After append(60):", list1)
list1.insert(1, 25)
print("After insert(1,25):", list1)
list1.remove(35)
print("After remove(35):", list1)
print("Length of list:", len(list1))
list1.sort()
print("Sorted List:", list1)
list1.reverse()
print("Reversed List:", list1)
