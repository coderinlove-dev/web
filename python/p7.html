Task / Question
7.To study function definition, calling functions, different types of arguments, scope of variables, and documentation strings.

# Function Definition and Documentation String
def greet(name, msg="Hello"):
    """Function to greet a person with a message"""
    print(f"{msg}, {name}!")

# Calling Functions
greet("Bhavani")
greet("Arya", "Good Morning")

# Different Types of Arguments
def calculate(a, b=5, *args, **kwargs):
    """Function to demonstrate different types of arguments"""
    print("a:", a)
    print("b:", b)
    print("args:", args)
    print("kwargs:", kwargs)
    return a + b + sum(args) + sum(kwargs.values())

result = calculate(10, 20, 1, 2, 3, x=5, y=10)
print("Result of calculate:", result)

# Scope of Variables
global_var = 100

def scope_demo():
    local_var = 50
    global global_var
    global_var += 1
    print("Inside function - Local:", local_var, "Global:", global_var)

scope_demo()
print("Outside function - Global:", global_var)

# Accessing Documentation Strings
print("\nFunction Documentation:")
print(greet.__doc__)
print(calculate.__doc__)
